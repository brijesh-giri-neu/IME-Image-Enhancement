#load a png image file 
load Image.png resImage3

# blur
blur resImage3 blurredImage1 split 50
blur blurredImage1 blurredImage2 split 50
blur blurredImage2 blurredImage3 split 50
save res/A5/blur-3times-Split50-Image.jpg blurredImage3

# sharpen
sharpen resImage3 resImageSharpen split 50
save res/A5/sharpen-1time-Split50-Image.jpg resImageSharpen

# sepia
sepia resImage3 resImageSepia split 50
save res/A5/sepia-Split50-Image.jpg resImageSepia

# Get luma-component - Grayscale command
luma-component resImage3 lumaResImage split 50
save res/A5/luma-Split50-Image.jpg lumaResImage

# Generate histogram of original Image
histogram resImage3 histResImage3
save res/A5/histogram-Image.jpg histResImage3

# Color correct 
color-correct resImage3 colorCorrectResImage
save res/A5/colorCorrect-Image.jpg colorCorrectResImage

# Generate histogram of color corrected Image
histogram colorCorrectResImage histcolorCorrectResImage
save res/A5/histogram-colorCorrect-Image.jpg histcolorCorrectResImage
